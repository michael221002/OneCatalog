// Definieren einer Klasse "Person"
class Person {
  // Definieren der Eigenschaften (Daten)
  name: string;
  age: number;

  // Definieren des Konstruktors, der die Eigenschaften initialisiert
  constructor(name: string, age: number) {
    this.name = name;
    this.age = age;
  }

  // Definieren einer Methode (Verhalten)
  sayHello() {
    console.log(`Hello, my name is ${this.name} and I am ${this.age} years old.`);
  }
}

// Erstellen einer Instanz der Klasse "Person"
const person1 = new Person('Alice', 25);

// Aufrufen der Methode "sayHello"
person1.sayHello();
