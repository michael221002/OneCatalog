interface Car {
  make: string;
  model: string;
  year?: number;
  color!: string;
  features: {
    powerWindows: boolean;
    sunroof: boolean;
  };
}

interface RequiredCar {
  make: string;
  model: string;
  year: number;
  color: string;
  features: {
    powerWindows: boolean;
    sunroof: boolean;
  };
}

interface PartialCar {
  make?: string;
  model?: string;
  year?: number;
  color?: string;
  features?: {
    powerWindows?: boolean;
    sunroof?: boolean;
  };
}

myCar: Car = {
    make: 'Toyota',
    model: 'Camry',
    year: 2022,
    color: 'blue',
    features: {
      powerWindows: true,
      sunroof: false,
    },
};
